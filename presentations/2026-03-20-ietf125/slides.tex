\begin{frame}{\color{gray}Why?}\Large\color{gray}
  GET /.well-known/core has 17 bytes of Uri-Path.

  \bigskip

  EDHOC was unhappy.

  \bigskip

  cBRSKI is unhappy.

  \bigskip

  IETFers are jealous of other SDOs\footnote{LwM2M's \texttt{/rd} is a BCP190 no-go}
\end{frame}

\begin{frame}{\color{gray}Solution}\Large\color{gray}
  Numeric Uri-Path-Abbrev option.

  \bigskip

  \begin{tabular}{l|l}
    Abbreviated & Expanded \\ \hline
    Uri-Path: 0 \hspace{1cm} & Uri-Path: ``.well-known'', ``core'' \\
    \texttt{d0 00} & \texttt{bb . w e l l - k n o w n 04 c o r e} \\
    2 & 17 \\
  \end{tabular}

  \bigskip

  Critical. Safe-to-forward\footnote{even though Uri-Path is not}.

  \bigskip

  Uri-Host (3), Uri-Port (7), \textbf{Uri-Path-Abbrev (13)}, Uri-Query (15).
\end{frame}

\begin{frame}{Open items after IETF124}\Large
  \framesubtitle{Part II: Simplify Proxy-Uri intraction}

  PRs \href{https://github.com/core-wg/uri-path-abbrev/issues/33}{\#33} and \href{https://github.com/core-wg/uri-path-abbrev/issues/38}{\textit{\#38}}\footnote{\label{yester}My apologies for the short-notice changes -- but presenting them in a single change is still better than giving you half of it here and half of it in the next interim.}
  
  \begin{itemize}
    \item Combinations of Uri-Path and Uri-Path-Abbrev are disallowed even when proxies introduce them through otherwise legal\footnote{Ongoing corr-clar issue \href{https://github.com/core-wg/corrclar/issues/51}{\#51}.} operations.

      Servers can just not support UPA when their reverse proxy meddles with the path.
      % no use cases, needless complication.
    \item \textit{As the last hop is not to a forward-proxy, servers will never see this.}
    \item \textit{Thus there is no general SHOULD for this case, and just a MUST for server-proxy hybrids}
  \end{itemize}
\end{frame}
